TypeScript - API Client Factory 
// lib/api-client.ts - Retry logic with backoff
class APIClient {
  constructor(baseURL: string, apiKey: string) {
    this.baseURL = baseURL;
    this.apiKey = apiKey;
  }

  async fetch<T>(endpoint: string, retryCount = 0): Promise<T> {
    try {
      const response = await fetch(\`\${this.baseURL}\${endpoint}\`, {
        headers: {
          'Authorization': \`Bearer \${this.apiKey}\`,
          'Content-Type': 'application/json',
        },
      });

      if (!response.ok) throw new Error(\`API Error: \${response.status}\`);
      return await response.json();
    } catch (error) {
      if (retryCount < 3) {
        await new Promise(r => setTimeout(r, 1000 * Math.pow(2, retryCount)));
        return this.fetch<T>(endpoint, retryCount + 1);
      }
      throw error;
    }
  }
}

-- Complete analytics schema for D1
CREATE TABLE worker_requests (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  worker_name TEXT NOT NULL,
  method TEXT NOT NULL,
  path TEXT NOT NULL,
  status_code INTEGER NOT NULL,
  duration_ms INTEGER NOT NULL,
  timestamp DATETIME DEFAULT CURRENT_TIMESTAMP,
  country TEXT
);

CREATE TABLE api_calls (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  service TEXT NOT NULL,
  endpoint TEXT NOT NULL,
  status_code INTEGER,
  duration_ms INTEGER,
  cost_usd DECIMAL(10, 6),
  success BOOLEAN NOT NULL,
  timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_worker_time ON worker_requests(timestamp);
CREATE INDEX idx_api_service ON api_calls(service, timestamp);
